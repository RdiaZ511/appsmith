SELECT 
    jsonb_build_object(
        'type', 'FeatureCollection',
        'features', jsonb_agg(feature)
    )
FROM (
    SELECT 
        jsonb_build_object(
            'type', 'Feature',
            'geometry', ST_AsGeoJSON(geom)::jsonb,  -- ← Quitar ST_Centroid
            'properties', jsonb_build_object(
                'parroquia', parroquia,
                'municipio', municipio
            )
        ) AS feature
    FROM parroquias_carabobo
    WHERE municipio = {{alcaldia.selectedOptionValue}}
) row;
