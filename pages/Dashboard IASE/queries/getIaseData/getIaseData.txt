SELECT
  sector,
  municipio,
  tamano,
  total_entidades,
  antiguedad_promedio,
  clasificacion,
  cnt_emergente,
  cnt_consolidacion,
  cnt_establecida,
  cnt_madura,
  cnt_ancla,
  operativa_promedio
FROM v_iase
WHERE ( {{!sectorFilter.selectedOptionValue}} OR sector = {{sectorFilter.selectedOptionValue}} )
  AND ( {{!municipioFilter.selectedOptionValue}} OR municipio = {{municipioFilter.selectedOptionValue}} )
  AND ( {{!tamanoFilter.selectedOptionValue}} OR tamano = {{tamanoFilter.selectedOptionValue}} )
ORDER BY municipio, antiguedad_promedio DESC;
